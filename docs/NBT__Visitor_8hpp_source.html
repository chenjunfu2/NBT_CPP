318protected:
320 struct Frame
321 {
323 enum class Type : uint8_t
324 {
325 Compound,
326 List,
327 };
328
329 public:
330 Type enType;
331 union
332 {
333 NBT_Type::Compound *pCompound;
334 NBT_Type::List *pList;
335 };
336 };

339 NBT_Type::String sPendingKey;
340 std::vector<Frame> vStack{};

349 template<typename T>
350 bool AppendStackTop(T &&tVal)
351 {
352 if (vStack.empty())
353 {
354 return false;
355 }
356
357 void *pNewElement = NULL;
358
359 Frame &stTopFrame = vStack.back();
360 switch (stTopFrame.enType)
361 {
362 case Frame::Type::Compound:
363 {
364 if (stTopFrame.pCompound == NULL)
365 {
366 return false;
367 }
368
369 //尝试插入，遇到重复则替换
370 auto [it, b] = stTopFrame.pCompound->Put(std::move(sPendingKey), std::forward<T>(tVal));
371 pNewElement = (void*)&(it->second);
372 }
373 break;
374 case Frame::Type::List:
375 {
376 if (stTopFrame.pList == NULL)
377 {
378 return false;
379 }
380
381 pNewElement = (void*)&stTopFrame.pList->AddBack(std::forward<T>(tVal));
382 }
383 break;
384 default:
385 return false;
386 break;
387 }
388
389 if constexpr (NBT_Type::IsCompoundType_V<T>)
390 {
391 vStack.push_back(
392 Frame
393 {
394 .enType = Frame::Type::Compound,
395 .pCompound = (NBT_Type::Compound *)pNewElement,
396 }
397 );
398 }
399 else if constexpr (NBT_Type::IsListType_V<T>)
400 {
401 vStack.push_back(
402 Frame
403 {
404 .enType = Frame::Type::List,
405 .pList = (NBT_Type::List *)pNewElement,
406 }
407 );
408 }
409
410 return true;
411 }
412
416 bool PopStack(Frame::Type enTypeRequires)
417 {
418 if (vStack.empty())
419 {
420 return false;
421 }
422
423 if (vStack.back().enType != enTypeRequires)
424 {
425 return false;
426 }
427
428 vStack.pop_back();
429 return true;
430 }
