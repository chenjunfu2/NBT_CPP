147 std::partial_ordering operator<=>(const NBT_Compound &_Right) const noexcept
148 {
149 if (auto _cmpSize = Compound::size() <=> _Right.size(); _cmpSize != 0)
150 {
151 return _cmpSize;
152 }
153
154 //数量相等，比较数据的排序
155 const auto _lSort = this->KeySortIt();
156 const auto _rSort = _Right.KeySortIt();
157 typename Compound::size_type _Size = Compound::size();
158
159 for (typename Compound::size_type _i = 0; _i < _Size; ++_i)
160 {
161 const auto &_lIt = _lSort[_i];
162 const auto &_rIt = _rSort[_i];
163
164 //首先比较名称，如果不同则返回
165 if (auto _cmpKey = _lIt->first <=> _rIt->first; _cmpKey != 0)
166 {
167 return _cmpKey;
168 }
169
170 //然后比较值，如果不同则返回
171 if (auto _cmpVal = _lIt->second <=> _rIt->second; _cmpVal != 0)
172 {
173 return _cmpVal;
174 }
175 }
176
177 //前面都没有返回，那么所有元素都相等
178 return std::partial_ordering::equivalent;
179 }
