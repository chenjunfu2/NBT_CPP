200 .next_in = Z_NULL,

202 .total_in = 0,

207

209 .state = Z_NULL,

212 .zfree = (free_func)Z_NULL,
213 .opaque = (voidpf)Z_NULL,
214
215 .data_type = Z_BINARY,
216
217 .adler = 0,
218 .reserved = {},
219 };

221 if (inflateInit2(&zs, 32 + 15) != Z_OK)//32+15自动判断是gzip还是zlib
222 {
223 throw std::runtime_error("Failed to initialize zlib decompression");
224 }
225
226 //对于大于uint_max的数据来说，切分为多个uint_max的块作为流依次输入
227 //注意zlib在实现内会移动指针，所以对于大于一定字节的情况下，只需要更新
228 //avail_in，而无须更新next_in。这里的容器保证不变，所以地址不会失效
229 zs.next_in = (z_const Bytef *)iData.data();
230
231 //默认解压大小为压缩大小，后续扩容
232 oData.resize(iData.size());
233
234 //两个变量用于记录已经压缩的大小和剩余数据大小
235 size_t szDecompressedSize = 0;
236 size_t szRemainingSize = iData.size();
237 int iRet = Z_OK;
238 do
239 {
240 //首先计算剩余可用空间，然后决定是否扩容
241 size_t szOut = oData.size() - szDecompressedSize;
242 if (szOut == 0)
243 {
244 oData.resize(oData.size() * 2);
245 szOut = oData.size() - szDecompressedSize;
246 }
247
248 //虽然next_out也会在内部实现被移动，但是oData是容器，会被扩容
249 //一旦扩容触发，那么实际上的地址就可能会改变，所以必须每次重新获取
250 //切记上面与这里的代码顺序不可改变，必须在上面计算并扩容完成后获取地址
251 zs.next_out = (Bytef *)(&oData.data()[szDecompressedSize]);
252
253 //如果输入被消耗完，重新赋值
254 if (zs.avail_in == 0)
255 {
256 constexpr uInt uIntMax = (uInt)-1;
257 zs.avail_in = szRemainingSize > (size_t)uIntMax ? uIntMax : (uInt)szRemainingSize;
258 szRemainingSize -= zs.avail_in;//缩小剩余待处理大小
259 }
260
261 //如果输出大小耗尽，重新赋值
262 if (zs.avail_out == 0)
263 {
264 constexpr uInt uIntMax = (uInt)-1;
265 zs.avail_out = szOut > (size_t)uIntMax ? uIntMax : (uInt)szOut;
266 //这里不对szOut处理，因为会在开头与结尾计算
267 }
268
269 //解压，如果剩余不为0则代表分块了，使用Z_NO_FLUSH，否则Z_FINISH
270 iRet = inflate(&zs, szRemainingSize != 0 ? Z_NO_FLUSH : Z_FINISH);
271
272 //计算本次解压的大小
273 szDecompressedSize += szOut - zs.avail_out;
274 } while (iRet == Z_OK || iRet == Z_BUF_ERROR);//只要没错误（缓冲区不够大除外）或者没到结尾就继续运行
275
276 inflateEnd(&zs);//结束解压
277 oData.resize(szDecompressedSize);//设置解压大小
278
279 //错误处理
280 if (iRet != Z_STREAM_END)
281 {
282 if (zs.msg != NULL)//错误消息不为null则抛出带消息异常
283 {
284 throw std::runtime_error(std::string("Zlib decompression failed with error message: ") + std::string(zs.msg));
285 }
286 else//如果为null直接使用错误码
287 {
288 throw std::runtime_error(std::string("Zlib decompression failed with error code: ") + std::to_string(iRet));
289 }
290 }
291 }

348 /*
349 如果范围溢出，则设置为比原始数据的大小加12字节大0.1%的一半
350 这样做的目的是为了尽可能缩小一开始的体积
351 比如实际上压缩率非常低的情况下可能根本用不到一半
352 但是如果实际上压缩率很高，那么下面二倍扩容一次后刚好就是最坏情况
353 这样基本上性能较优，下面zlib文档说明的最坏情况：
354
355 Upon entry, destLen is the total size of the
356 destination buffer, which must be at least 0.1%
357 larger than sourceLen plus 12 bytes.
358 */
359
360 constexpr uLong uLongMax = (uLong)-1;
361 if (iData.size() > (size_t)uLongMax)
362 {
363 size_t szNeedSize = iData.size() + 12;//先比原始数据大12byte
364 //注意这里使用了向上取整的整数除法
365 //加等于自身的0.1%相当于比原先的自己大0.1%，这里的1/1000就是0.1/100
366 szNeedSize += (szNeedSize + (1000 - 1)) / 1000;
367 //设置目标大小
368 oData.resize((szNeedSize + (2 - 1)) / 2);//向上取整除以二
369 }
370 else
371 {
372 //在范围未溢出的情况下，进行预测
373 //把压缩大小设置为预测的压缩大小
374 oData.resize(deflateBound(&zs, (uLong)iData.size()));
375 }
376
377 //设置压缩后大小与待处理大小
378 size_t szCompressedSize = 0;
379 size_t szRemainingSize = iData.size();
380 int iRet = Z_OK;
381 do
382 {
383 //计算剩余大小并在不足时扩容
384 size_t szOut = oData.size() - szCompressedSize;
385 if (szOut == 0)
386 {
387 oData.resize(oData.size() * 2);
388 szOut = oData.size() - szCompressedSize;
389 }
390
391 //获取新的地址
392 zs.next_out = (Bytef *)(&oData.data()[szCompressedSize]);
393
394 //如果输入被消耗完，重新赋值
395 if (zs.avail_in == 0)
396 {

399 szRemainingSize -= zs.avail_in;//缩小剩余待处理大小

401
