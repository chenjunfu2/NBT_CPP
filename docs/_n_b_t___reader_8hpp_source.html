245 template <typename T, typename InputStream, typename ErrInfoFunc, typename... Args>
246 requires(std::is_same_v<T, ErrCode> || std::is_same_v<T, WarnCode>)
247 static std::conditional_t<std::is_same_v<T, ErrCode>, ErrCode, void> Error
248 (
249 const T &code,
250 const InputStream &tData,
251 ErrInfoFunc &funcErrInfo,
252 const std::FMT_STR<Args...> fmt,
253 Args&&... args
254 ) noexcept
255 {
256 //打印错误原因
257 if constexpr (std::is_same_v<T, ErrCode>)
258 {
259 if (code >= ERRCODE_END)
260 {
261 return code;
262 }
263 //上方if保证code不会溢出
264 funcErrInfo("Read Err[{}]: {}\n", (uint8_t)code, errReason[code]);
265 }
266 else if constexpr (std::is_same_v<T, WarnCode>)
267 {
268 if (code >= WARNCODE_END)
269 {
270 return;
271 }
272 //上方if保证code不会溢出
273 funcErrInfo("Read Warn[{}]: {}\n", (uint8_t)code, warnReason[code]);
274 }
275 else
276 {
277 static_assert(false, "Unknown [T code] Type!");
278 }
279
280 //打印扩展信息
281 funcErrInfo("Extra Info: \"");
282 funcErrInfo(std::move(fmt), std::forward<Args>(args)...);
283 funcErrInfo("\"\n\n");
284
285 //如果可以，预览szCurrent前后n个字符，否则裁切到边界
287#define VIEW_PRE (4 * 8 + 3)//向前
288#define VIEW_SUF (4 * 8 + 5)//向后
290 size_t rangeBeg = (tData.Index() > VIEW_PRE) ? (tData.Index() - VIEW_PRE) : (0);//上边界裁切
291 size_t rangeEnd = ((tData.Index() + VIEW_SUF) < tData.Size()) ? (tData.Index() + VIEW_SUF) : (tData.Size());//下边界裁切
292#undef VIEW_SUF
293#undef VIEW_PRE
294 //输出信息
295 funcErrInfo
296 (
297 "Data Review:\n"\
298 "Current: 0x{:02X}({})\n"\
299 "Data Size: 0x{:02X}({})\n"\
300 "Data Range: [0x{:02X}({}),0x{:02X}({})):\n",
301
302 (uint64_t)tData.Index(), tData.Index(),
303 (uint64_t)tData.Size(), tData.Size(),
304 (uint64_t)rangeBeg, rangeBeg,
305 (uint64_t)rangeEnd, rangeEnd
306 );
307
308 //打数据
309 for (size_t i = rangeBeg; i < rangeEnd; ++i)
310 {
311 if ((i - rangeBeg) % 8 == 0)//输出地址
312 {
313 if (i != rangeBeg)//除去第一个每8个换行
314 {
315 funcErrInfo("\n");
316 }
317 funcErrInfo("0x{:02X}: ", (uint64_t)i);
318 }
319
320 if (i != tData.Index())
321 {
322 funcErrInfo(" {:02X} ", (uint8_t)tData[i]);
323 }
324 else//如果是当前出错字节，加方括号框起
325 {
326 funcErrInfo("[{:02X}]", (uint8_t)tData[i]);
327 }
328 }
329
330 //输出提示信息
331 if constexpr (std::is_same_v<T, ErrCode>)
332 {
333 funcErrInfo("\nSkip err data and return...\n\n");
334 }
335 else if constexpr (std::is_same_v<T, WarnCode>)
336 {
337 funcErrInfo("\nSkip warn data and continue...\n\n");
338 }
339 else
340 {
341 static_assert(false, "Unknown [T code] Type!");
342 }
343
344 //警告不返回值
345 if constexpr (std::is_same_v<T, ErrCode>)
346 {
347 return code;
348 }
349 }

395 requires std::integral<T>
396 static inline std::conditional_t<bNoCheck, void, ErrCode> ReadBigEndian(InputStream &tData, T &tVal, ErrInfoFunc &funcErrInfo) noexcept
397 {
398 if constexpr (!bNoCheck)
399 {
400 if (!tData.HasAvailData(sizeof(T)))
401 {
402 ErrCode eRet = Error(OutOfRangeError, tData, funcErrInfo, "tData size [{}], current index [{}], remaining data size [{}], but try to read [{}]",
403 tData.Size(), tData.Index(), tData.Size() - tData.Index(), sizeof(T));
404 STACK_TRACEBACK("HasAvailData Test");
405 return eRet;
406 }
407 }
408
409 T BigEndianVal{};
410 tData.GetRange((uint8_t *)&BigEndianVal, sizeof(BigEndianVal));
411 tVal = NBT_Endian::BigToNativeAny(BigEndianVal);
412
413 if constexpr (!bNoCheck)
414 {
415 return AllOk;
416 }
417 }

515 template<bool bRoot, typename InputStream, typename ErrInfoFunc>
516 static ErrCode GetCompoundType(InputStream &tData, NBT_Type::Compound &tCompound, size_t szStackDepth, ErrInfoFunc &funcErrInfo) noexcept
517 {
518 MYTRY;
519 ErrCode eRet = AllOk;
520 CHECK_STACK_DEPTH(szStackDepth);
521
522 //读取
523 while (true)
524 {
525 //处理末尾情况
526 if (!tData.HasAvailData(sizeof(NBT_TAG_RAW_TYPE)))
527 {
528 if constexpr (!bRoot)//非根部情况遇到末尾，则报错
529 {
530 eRet = Error(OutOfRangeError, tData, funcErrInfo, "{}:\nIndex[{}] >= DataSize()[{}]", __FUNCTION__,
531 tData.Index(), tData.Size());
532 }
533
534 return eRet;//否则直接返回（默认值AllOk）
535 }
536
537 //先读取一下类型
538 NBT_TAG tagNbt = (NBT_TAG)(NBT_TAG_RAW_TYPE)tData.GetNext();
539 if (tagNbt == NBT_TAG::End)//处理End情况
540 {
541 return eRet;//直接返回（默认值AllOk）
542 }
543
544 if (tagNbt >= NBT_TAG::ENUM_END)//确认在范围内
545 {
546 eRet = Error(NbtTypeTagError, tData, funcErrInfo, "{}:\nNBT Tag switch default: Unknown Type Tag[0x{:02X}({})]", __FUNCTION__,
547 (NBT_TAG_RAW_TYPE)tagNbt, (NBT_TAG_RAW_TYPE)tagNbt);//此处不进行提前返回，往后默认返回处理
548 STACK_TRACEBACK("tagNbt Test");
549 return eRet;//超出范围立刻返回
550 }
551
552 //然后读取名称
553 NBT_Type::String sName{};
554 eRet = GetName(tData, sName, funcErrInfo);
555 if (eRet != AllOk)
556 {
557 STACK_TRACEBACK("GetName Fail, Type: [NBT_Type::{}]", NBT_Type::GetTypeName(tagNbt));
558 return eRet;//名称读取失败立刻返回
559 }
560
561 //然后根据类型，调用对应的类型读取并返回到tmpNode
562 NBT_Node tmpNode{};
563 eRet = GetSwitch(tData, tmpNode, tagNbt, szStackDepth - 1, funcErrInfo);
564 if (eRet != AllOk)
565 {
566 STACK_TRACEBACK("GetSwitch Fail, Name: \"{}\", Type: [NBT_Type::{}]", sName.ToCharTypeUTF8(), NBT_Type::GetTypeName(tagNbt));
567 //return eRet;//注意此处不返回，进行插入，以便分析错误之前的正确数据
568 }
569
570 //sName:tmpNode，插入当前调用栈深度的根节点
571 //根据实际mc java代码得出，如果插入一个已经存在的键，会导致原先的值被替换并丢弃
572 //那么在失败后，手动从迭代器替换当前值，注意，此处必须是try_emplace，因为try_emplace失败后原先的值
573 //tmpNode不会被移动导致丢失，所以也无需拷贝插入以防止移动丢失问题
574 auto [it, bSuccess] = tCompound.try_emplace(std::move(sName), std::move(tmpNode));
575 if (!bSuccess)
576 {
577 //使用当前值替换掉阻止插入的原始值
578 it->second = std::move(tmpNode);
579
580 //发出警告，注意警告不用eRet接返回值
581 Error(ElementExistsWarn, tData, funcErrInfo, "{}:\nName: \"{}\", Type: [NBT_Type::{}] data already exist!", __FUNCTION__,
582 sName.ToCharTypeUTF8(), NBT_Type::GetTypeName(tagNbt));
583 }
584
585 //最后判断是否出错
586 if (eRet != AllOk)
587 {
588 STACK_TRACEBACK("While break with an error!");
589 return eRet;//出错返回
590 }
591 }
592
593 return eRet;//返回错误码
594 MYCATCH;
595 }

615 {
616 MYTRY;
617 ErrCode eRet = AllOk;

634 STACK_TRACEBACK("enListElementTag Test");
635 return eRet;
636 }
637
638 //读取4字节的有符号列表长度
639 NBT_Type::ListLength iListLength = 0;//4byte

645 }

656 if (enListElementTag == NBT_TAG::End && iListLength != 0)

809 备注：此函数读取nbt时，会创建一个默认根，然后把nbt内所有数据集合到此默认根上，
810 也就是哪怕按照mojang的nbt标准，默认根是无名Compound，也会被挂接到返回值里的
811 NBT_Type::Compound中。遍历函数返回的NBT_Type::Compound即可得到所有NBT数据，
812 这么做的目的是为了方便读写例程且不用在某些地方部分实现mojang的无名compound的
813 特殊处理，这种情况下可以在一定程度上甚至比mojang标准支持更多的NBT文件情况，
814 比如文件内并不是Compound开始的，而是单纯的几个不同类型且带有名字的NBT，那么也能
815 正常读取到并全部挂在NBT_Type::Compound中，就好像nbt文件本身就是一个大的无名
816 NBT_Type::Compound一样，相对的，写出函数也能支持写出此种情况，所以写出函数
817 WriteNBT在写出的时候，传入的值也是一个内含NBT_Type::Compound的NBT_Node，
818 然后传入的NBT_Type::Compound本身不会被以任何形式写入NBT文件，而是内部数据，
819 也就是挂接在下面的内容会被写入，这样既能保证兼容mojang的nbt文件，也能一定程度上
820 扩展nbt文件内可以存储的内容（允许nbt文件直接存储多个键值对而不是必须先挂在一个
821 无名称的Compound下）
822 */
